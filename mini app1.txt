<script>
function submitTransfer() {
  const owner = document.getElementById("ownerId").value;
  const newOwner = document.getElementById("newOwnerId").value;
  const plate = document.getElementById("plate").value;
  const type = document.getElementById("vehicleType").value;

  if (!owner || !newOwner || !plate || type === "اختر نوع المركبة") {
    alert("يرجى تعبئة جميع البيانات المطلوبة");
    return;
  }

  document.getElementById("msg").innerText = "✅ تم نقل الملكية بنجاح (نسخة خفيفة)";
  setTimeout(() => window.location.href = "index.html", 2500);
}
